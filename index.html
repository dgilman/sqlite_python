conn = sqlite3.connect("db.sqlite3")
cur = conn.cursor()

   cur.execute("""
      SELECT actor_id, actor_name
      FROM actor
      WHERE actor_type = ?""", ("Stage",))
   for actor_id, actor_name in cur:
      print(f"{actor_name} ({actor_id})")
